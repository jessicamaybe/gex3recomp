cmake_minimum_required(VERSION 3.20)
project(Gex3Recompiled LANGUAGES C CXX)

set(CMAKE_C_STANDARD 17)
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)

# Zelda-style defaults
set(RT64_STATIC TRUE CACHE BOOL "" FORCE)
set(RT64_SDL_WINDOW_VULKAN TRUE CACHE BOOL "" FORCE)
add_compile_definitions(HLSL_CPU)

# Subdirectories
# Note: N64ModernRuntime and RT64 are added as subdirectories.
# We need to make sure their internal paths (like ../thirdparty) work or fix them.
add_subdirectory(lib/N64ModernRuntime)
add_subdirectory(lib/RT64 ${CMAKE_BINARY_DIR}/rt64)

# RecompiledFuncs - Library containing the primary recompiler output
add_library(RecompiledFuncs STATIC)

target_compile_options(RecompiledFuncs PRIVATE
    -fno-strict-aliasing
    -Wno-unused-variable
    -Wno-implicit-function-declaration
    -w # Suppress warnings from generated code
)

if(CMAKE_SIZEOF_VOID_P EQUAL 8)
    target_compile_options(RecompiledFuncs PRIVATE -msse4.1 -mssse3)
endif()

target_include_directories(RecompiledFuncs PRIVATE
    ${CMAKE_SOURCE_DIR}/src
    ${CMAKE_SOURCE_DIR}/src/include
    ${CMAKE_SOURCE_DIR}/lib/N64Recomp/include
    ${CMAKE_SOURCE_DIR}/lib/N64ModernRuntime/ultramodern/include
    ${CMAKE_SOURCE_DIR}/lib/N64ModernRuntime/librecomp/include
    ${CMAKE_SOURCE_DIR}/lib/N64ModernRuntime/N64Recomp/include
    ${CMAKE_SOURCE_DIR}/RecompiledFuncs
)


# RecompiledFuncs are generated by running `N64Recomp gex3.toml` from the repo
# root (see BUILDING.MD). All recompiler corrections — fall-through function
# boundaries and the function text hooks that replaced the old fix_*.py scripts —
# live declaratively in gex3.toml / Gex3Syms/gex3.syms.toml, so no post-processing
# step is needed.
file(GLOB FUNC_C_SOURCES CONFIGURE_DEPENDS ${CMAKE_SOURCE_DIR}/RecompiledFuncs/funcs_*.c)
target_sources(RecompiledFuncs PRIVATE ${FUNC_C_SOURCES})

# Main executable
add_executable(gex3)

# Handle patches conditionally mimicking BanjoRecomp
file(GLOB PATCH_C_SOURCES ${CMAKE_SOURCE_DIR}/patches/*.c)

if(PATCH_C_SOURCES)
    if(NOT DEFINED PATCHES_C_COMPILER)
        set(PATCHES_C_COMPILER clang)
    endif()

    if(NOT DEFINED PATCHES_LD)
        set(PATCHES_LD ld.lld)
    endif()

    add_custom_target(PatchesElf
        COMMAND ${CMAKE_COMMAND} -E env CC=${PATCHES_C_COMPILER} LD=${PATCHES_LD} make
        WORKING_DIRECTORY ${CMAKE_SOURCE_DIR}/patches
        BYPRODUCTS ${CMAKE_SOURCE_DIR}/patches/patches.elf
    )

    add_custom_command(OUTPUT
                           ${CMAKE_SOURCE_DIR}/patches/patches.bin
                           ${CMAKE_SOURCE_DIR}/RecompiledPatches/patches.c
                       COMMAND ${CMAKE_SOURCE_DIR}/lib/N64Recomp/build/N64Recomp patches.toml
                       WORKING_DIRECTORY ${CMAKE_SOURCE_DIR}
                       DEPENDS ${CMAKE_SOURCE_DIR}/patches/patches.elf
    )

    target_sources(gex3 PRIVATE ${CMAKE_SOURCE_DIR}/RecompiledPatches/patches.c)
endif()

target_sources(gex3 PRIVATE
    src/app/main.cpp
    src/audio/gex_audio.cpp
    src/video/gex_rt64.cpp
    src/input/gex_input.cpp
    src/natives/gex_natives.cpp
    rsp/aspMain.cpp
    src/app/missing_traps.cpp
)

target_include_directories(gex3 PRIVATE
    ${CMAKE_SOURCE_DIR}/src
    ${CMAKE_SOURCE_DIR}/src/include
    ${CMAKE_SOURCE_DIR}/RecompiledFuncs
    ${CMAKE_SOURCE_DIR}/lib/N64Recomp/include
    ${CMAKE_SOURCE_DIR}/lib/N64ModernRuntime/ultramodern/include
    ${CMAKE_SOURCE_DIR}/lib/N64ModernRuntime/librecomp/include
    ${CMAKE_SOURCE_DIR}/lib/N64ModernRuntime/thirdparty
    ${CMAKE_SOURCE_DIR}/lib/N64ModernRuntime/thirdparty/xxHash
    ${CMAKE_SOURCE_DIR}/lib/N64ModernRuntime/thirdparty/concurrentqueue
    ${CMAKE_SOURCE_DIR}/lib/N64ModernRuntime/N64Recomp/include
    ${CMAKE_SOURCE_DIR}/lib/RT64/include
    ${CMAKE_SOURCE_DIR}/lib/RT64/src
    ${CMAKE_SOURCE_DIR}/lib/RT64/src/contrib
    ${CMAKE_SOURCE_DIR}/lib/RT64/src/contrib/hlslpp/include
    ${CMAKE_SOURCE_DIR}/lib/RT64/src/contrib/plume
)

if(CMAKE_SIZEOF_VOID_P EQUAL 8)
    target_compile_options(gex3 PRIVATE -msse4.1 -mssse3)
endif()

# SSE flags for RSP/Recomp logic in main app too
target_compile_definitions(gex3 PRIVATE XXH_INLINE_ALL)

# Dependencies
find_package(SDL2 REQUIRED)
target_link_libraries(gex3 PRIVATE SDL2::SDL2)

find_package(PkgConfig REQUIRED)
pkg_check_modules(GTK3 REQUIRED gtk+-3.0)
target_link_libraries(gex3 PRIVATE ${GTK3_LIBRARIES})
target_include_directories(gex3 PRIVATE ${GTK3_INCLUDE_DIRS})

target_link_libraries(gex3 PRIVATE
    RecompiledFuncs
    librecomp
    ultramodern
    rt64
    pthread
    dl
)

# Set runtime working directory (VS specific but helpful)
set_property(TARGET gex3 PROPERTY VS_DEBUGGER_WORKING_DIRECTORY "${CMAKE_SOURCE_DIR}/build")
